SQL Query 8:
SELECT *
FROM (
  SELECT
    AllData.*,
    SubStr("Período Apuração Crédito", 14, 4) AS Ano
  FROM AllData
) AS DadosComAno -- Alias para a subconsulta
WHERE Ano = '2019';

SQL Query 9:
SELECT
  AllData.*,
  SubStr("Período Apuração Crédito", 14, 4) AS Ano
FROM
  AllData
WHERE
  SubStr("Período Apuração Crédito", 14, 4) = '2019' -- Filtra primeiro por ano '2019'
ORDER BY
  Ano ASC,                         -- 1. Ordena pelo Ano calculado (crescente - ASC é o padrão, pode omitir)
  "Período Apuração Crédito" ASC,  -- 2. Desempata pelo período completo (crescente - ASC é o padrão)
  "Tipo Crédito" DESC;             -- 3. Desempata pelo tipo de crédito (decrescente)

SQL Query 10:
WITH DadosComAno AS (
  SELECT
    AllData.*,
    CAST(SubStr("Período Apuração Crédito", 14, 4) AS INTEGER) AS Ano
  FROM AllData
)
SELECT *
FROM DadosComAno
WHERE Ano > 2019 -- Filtra usando o alias da CTE
ORDER BY
  Ano ASC,                         -- 1. Ordena pelo Ano inteiro (crescente)
  "Período Apuração Crédito" ASC,  -- 2. Desempata pelo período completo (crescente)
  "Tipo Crédito" DESC;             -- 3. Desempata pelo tipo de crédito (decrescente)
